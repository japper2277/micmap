Questions that trigger this flow:

Code pattern:
```javascript
function handleLocationQuestion(question) {
  if (STATE.userLocation) {
    // Already have location - answer immediately
    return answerWithLocation(question);
  } else {
    // Need permission first
    return {
      message: "I need your location to find nearby mics",
      buttons: [{
        label: "Allow Location",
        action: async () => {
          await getUserLocation();
          if (STATE.userLocation) {
            return answerWithLocation(question);
          } else {
            return "Location access denied. Try searching by borough instead.";
          }
        }
      }, {
        label: "Search by borough",
        action: () => showBoroughOptions()
      }]
    };
  }
}
```
